\block{\bt{1.}動機と問い}{%
\begin{itemize}
\setlength{\itemsep}{1mm}
\item 株価そのものは\textbf{予測しない}
\item 市場の\textbf{``つながりの形''}を見る
\item 形の異変から下落を\textbf{\color{accent}先取り}する
\end{itemize}

\vspace{2mm}
\textbf{先行研究}:
\begin{itemize}
\setlength{\itemsep}{1mm}
\item Mantegna (1999)：相関 $\to$ \textbf{ネットワーク}
\item Gidea (2017)：\textbf{``形''} $\to$ 危機検知
\item Ferreira ら (2021)：\textbf{``符号''} $\to$ 危機検知
\end{itemize}

\vspace{1.5mm}
\centering
\colorbox{softblue}{\parbox{0.9\linewidth}{\centering
\textbf{``形''＋``符号'' の同時研究は {\color{warn}ない}}}}
\raggedright

\vspace{2mm}
\textbf{問い}:
\begin{itemize}
\setlength{\itemsep}{1mm}
\item 2指標は\textbf{独立}か？
\item 組み合わせて下落を\textbf{先取り}できるか？
\end{itemize}

\vspace{2mm}
{\color{muted}\small\textbf{日次リターン}＝前日終値からの変化率。
\textbf{相関 $r$}＝2銘柄が同じ向きに動くか（$+1$＝いつも同じ向き、$0$＝無関係、$-1$＝いつも逆向き）。}
}

\block{\bt{2.}相関ネットワーク}{%
\begin{itemize}
\setlength{\itemsep}{1mm}
\item \textbf{正相関}（一緒に動く）＝実線
\item \textbf{負相関}（逆に動く）＝破線
\item 相関 $r$ を距離 $d=\sqrt{2(1-r)}$ に変換して扱う（Mantegna 1999）
\end{itemize}

\vspace{2mm}
\centering
\includegraphics[width=0.82\linewidth]{../../figs/slide_network.png}\\
{\color{muted}\small \textbf{点の配置に意味はない}（見やすさのための模式図）。
指標は図でなく\textbf{相関行列から直接}計算する}

\vspace{2mm}
\raggedright
この\textbf{``形''}（3.1）と\textbf{``符号''}（3.2）から2指標を取り出す。
}

\block{\bt{3.1}指標1：穴の持続期間}{%
\begin{itemize}
\setlength{\itemsep}{1mm}
\item 各点に円を描き、\textbf{半径を大きく}する
\item 円が重なった対を\textbf{辺で結ぶ}
\item \textbf{``輪（穴）''}が現れ、やがて埋まって消える
\end{itemize}

\vspace{1mm}
\centering
\includegraphics[width=\linewidth]{../../figs/slide_hole.png}

\vspace{1mm}
各穴の\textbf{寿命}（消える半径 $-$ 現れる半径）を測る:

\vspace{1mm}
\includegraphics[width=\linewidth]{../../figs/slide_lifetime.png}

\vspace{1mm}
\raggedright
指標 $L^1$ ＝ すべての穴の寿命の合計:
\[
L^1 = \sum_{(b_k,d_k)\in\mathrm{Dgm}(H_1)}(d_k - b_k),\quad
\mathrm{Dgm}(H_1)=\{(b_k,d_k)\}
\]
{\color{muted}\small $\mathrm{Dgm}(H_1)$＝穴の一覧、$b_k$＝現れる半径、$d_k$＝消える半径。}
}

\column{0.333}

\block{\bt{3.2}指標2：不均衡サイクルの数}{%
各辺に\textbf{符号}：正相関＝$+$、負相関＝$-$。

\vspace{1mm}
\textbf{直感}：A--B が一緒、B--C も一緒 $\to$ \textbf{A--C も一緒のはず}。
なのに\textbf{逆}なら \textbf{\color{warn}矛盾（ねじれ）}:

\vspace{1.5mm}
\centering
\includegraphics[width=0.78\linewidth]{../../figs/slide_balance.png}

\vspace{1mm}
\raggedright
長いサイクル $C$ でも同じ：一周の\textbf{符号の積が $-$}（負相関が\textbf{奇数個}）$\to$ どこかに必ず矛盾＝\textbf{\color{warn}不均衡サイクル}。
\[
\nunb = \text{不均衡サイクルの個数}\quad\Bigl(\textstyle\prod_{\text{辺}\in C}\mathrm{sign}(r)=-1\Bigr)
\]

\vspace{2mm}
\centering
\textbf{実データの例}（2023年12月・30日、点＝1日の値動き）:\\[1mm]
\includegraphics[width=\linewidth]{../../figs/slide_balance_real.png}
}

\block{\bt{3.3}2つの指標の独立性}{%
穴と不均衡の\textbf{連動}を相関で測る:

\vspace{1mm}
\centering
\includegraphics[width=\linewidth]{../../figs/slide_scatter_raw.png}

\vspace{1mm}
\raggedright
\begin{itemize}
\setlength{\itemsep}{2mm}
\item \textbf{近年ほど\color{accent}独立}（直近5年 $r=0.15$）$\to$ \textbf{別々の情報}
\item 古いほど\textbf{\color{warn}連動}（2006--11年 0.45 $\to$ 直近 0.15）。20年通しの 0.40 は\textbf{2008年危機の寄与}
\end{itemize}
}

\block{\bt{3.4}2指標の統合：$\ediv$}{%
スケールが違う（$L^1$: 0〜2、$\nunb$: 0〜80）$\to$ \textbf{標準化 $z$} して引く:
\[
z(x_t) = \frac{x_t - \mathrm{mean}(x_{\le t})}{\mathrm{sd}(x_{\le t})}
\quad\Longrightarrow\quad
\boxed{\ediv = z(\nunb) - z(L^1)}
\]
{\color{muted}\small mean・sd はその日まで（未来を見ない）}

\vspace{1.5mm}
\begin{itemize}
\setlength{\itemsep}{1.5mm}
\item \textbf{\color{warn}$\ediv > 0$}：不均衡＞穴 ＝ 関係が\textbf{ねじれ始めた}
$\to$ \textbf{\color{warn}崩壊の兆し} $\to$ 株を避ける
\item \textbf{$\ediv < 0$}：穴＞不均衡 ＝ \textbf{まとまりが保たれている} $\to$ 株を持つ
\end{itemize}
}

\block{\bt{3.5}なぜ「差分」なのか}{%
片方だけでは効かない。同じ条件を\textbf{20年全体（2006--26年）に当てはめた}ときの最大下落（指標どうしの比較。未来を見ない\textbf{前進検証}は4章）:

\vspace{2mm}
\centering
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lcc}
\toprule
{\bfseries 使う指標} & {\bfseries 最大下落} & {\bfseries シャープ比}\\
\midrule
\textbf{$\ediv$（差分）} & {\color{accent}\textbf{$-20\%$}} & {\color{accent}\textbf{0.76}}\\
不均衡 $\nunb$ だけ & $-45\%$ & 0.47\\
穴 $L^1$ だけ & $-56\%$ & 0.22\\
何もしない & $-57\%$ & 0.48\\
\bottomrule
\end{tabular}

\vspace{2mm}
\raggedright
\begin{itemize}
\setlength{\itemsep}{1mm}
\item 片方だけでは\textbf{何もしないのと大差なし}
\item \textbf{\color{accent}差分にして初めて $-20\%$ まで改善}
\end{itemize}
}
